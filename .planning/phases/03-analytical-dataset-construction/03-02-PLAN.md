```sas
/* Step 1: demo_validEnroll already filtered for valid enrollment */
/* Step 2: Join with dx, group by ID, get first admit_date per cancer status */
proc sql;
create table demo_validEnroll_dx_2 as
select ID, ENCOUNTERID, any_reportable_cancer, admit_date,
min(admit_date) as first_admit_date format=date9.
from demo_validEnroll_dx_1_
group by ID, any_reportable_cancer;
quit;

/* Step 3: Sort by id, descending any_reportable_cancer, admit_date -> keep first per id */
proc sort data=demo_validEnroll_dx_3; by id descending any_reportable_cancer admit_date; run;
proc sort data=demo_validEnroll_dx_3 nodupkey; by id; run;

/* Step 4: Flag patients with reportable cancer */
if any_reportable_cancer=1 then first_admit_date_forcancer=1;
else first_admit_date_forcancer=0;

/* Step 5: Keep only patients with cancer */
where first_admit_date_forcancer=1;
```
