# sdif is a range-based format (SAS value ranges, not discrete codes)
# Translate as a function that can be used in mutate():
sas_formats$sdif <- list(
  breaks = c(0, 45, 73, 100),
  labels = c("SDI_1stT_0_45", "SDI_2ndT_46_73", "SDI_3rdT_74_100"),
  type = "range",
  apply = function(x) {
    cut(x, breaks = c(-Inf, 45, 73, Inf),
        labels = c("SDI_1stT_0_45", "SDI_2ndT_46_73", "SDI_3rdT_74_100"),
        right = TRUE)
